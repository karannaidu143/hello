module adder(A,B,C,D,Y);
input A,B,C,D;
output Y;
wire and_opl, and_op2;
and gl(and_opl,A,B); // g1 upper AND
and g2(and_op2,C,D); // g2 represents lower AND
or g3(Y,and_opl,and_op2); // g3 represents the OR gate
endmodule
